## Error & Rescue Registry

| Failure Mode | Recovery Strategy |
|-------------|-------------------|
| GitHub API 403 (private repo) | Show "This repo is private. Set GITHUB_TOKEN in environment for access." |
| GitHub API 429 (rate limit) | Show "GitHub rate limit reached. Set GITHUB_TOKEN for 5000 req/hr." with retry timer |
| GitHub API 404 (not found) | Show "Repository not found. Check the URL and try again." |
| Form save missing description | Description is optional in AgentProfile — graceful if empty |
| AI assist returns empty canUseToolPolicy | Default to `{ autoApprove: [], autoDeny: [] }` — already handled in `toAssistResult` |

Add above the `headers()` function:

```typescript
export class GitHubApiError extends Error {
  constructor(
    message: string,
    public status: number,
    public isPrivate: boolean = false,
    public isRateLimit: boolean = false,
    public isNotFound: boolean = false,
    public retryAfterMs?: number,
  ) {
    super(message);
    this.name = "GitHubApiError";
  }
}

function toGitHubError(res: Response, context: string): GitHubApiError {
  const status = res.status;
  if (status === 404) {
    return new GitHubApiError(
      `${context}: repository not found. Check the URL and try again.`,
      404, false, false, true,
    );
  }
  if (status === 403 || status === 401) {
    const isRateLimit = res.headers.get("x-ratelimit-remaining") === "0";
    if (isRateLimit) {
      const resetEpoch = Number(res.headers.get("x-ratelimit-reset") || 0);
      const retryAfterMs = resetEpoch ? (resetEpoch * 1000 - Date.now()) : 60_000;
      return new GitHubApiError(
        `GitHub rate limit reached. ${process.env.GITHUB_TOKEN ? "Try again later." : "Set GITHUB_TOKEN for 5,000 requests/hour."}`,
        status, false, true, false, retryAfterMs,
      );
    }
    return new GitHubApiError(
      `${context}: access denied. ${process.env.GITHUB_TOKEN ? "Check your token permissions." : "This may be a private repo — set GITHUB_TOKEN for access."}`,
      status, true, false, false,
    );
  }
  return new GitHubApiError(`${context}: GitHub API error ${status}`, status);
}
```

```typescript
export async function getDefaultBranch(owner: string, repo: string): Promise<string> {
  const res = await fetch(`${GITHUB_API}/repos/${owner}/${repo}`, {
    headers: headers(),
  });
  if (!res.ok) {
    throw toGitHubError(res, `Failed to access ${owner}/${repo}`);
  }
  const data = await res.json();
  return data.default_branch ?? "main";
}
```

```typescript
export async function getRepoTree(
  owner: string,
  repo: string,
  sha: string
): Promise<TreeEntry[]> {
  const res = await fetch(
    `${GITHUB_API}/repos/${owner}/${repo}/git/trees/${sha}?recursive=1`,
    { headers: headers() }
  );
  if (!res.ok) {
    throw toGitHubError(res, `Failed to read repository tree`);
  }
  const data = await res.json();
  return data.tree ?? [];
}

export async function getFileContent(
  owner: string,
  repo: string,
  path: string,
  ref: string
): Promise<string> {
  const url = `${RAW_BASE}/${owner}/${repo}/${ref}/${path}`;
  const res = await fetch(url);
  if (!res.ok) {
    throw toGitHubError(res, `Failed to fetch ${path}`);
  }
  return res.text();
}
```

```typescript
export async function getLatestCommitSha(
  owner: string,
  repo: string,
  branch: string
): Promise<string> {
  const res = await fetch(
    `${GITHUB_API}/repos/${owner}/${repo}/git/ref/heads/${branch}`,
    { headers: headers() }
  );
  if (!res.ok) {
    throw toGitHubError(res, `Failed to get branch '${branch}'`);
```
